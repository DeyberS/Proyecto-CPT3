  <script>
    $(document).ready(function() {
      // 1. Cambiar entre modos
      $('#observaciones').on('change', function() {
        if ($(this).val() === 'Despacho a Paciente') {
          $('#seccion_manual').hide(); $('#seccion_despacho').fadeIn();
        } else {
          $('#seccion_despacho').hide(); $('#seccion_manual').fadeIn();
        }
      });

      // 2. Cargar lotes para salida MANUAL
      $('#Id_descripcion_medicamento').on('change', function() {
        const id = $(this).val();
        if(id) {
          $.ajax({
            url: '../../cfg/ajax/obtener_descripcion_medicamento.php',
            type: 'POST', data: { id: id }, dataType: 'json',
            success: function(data) {
              let html = '<option value="">-- Seleccione Lote --</option>';
              data.lotes.forEach(l => {
                html += `<option value="${l.lote}">${l.lote} (Vence: ${l.fecha_vencimiento} | Stock: ${l.cantidad_actual})</option>`;
              });
              $('#lista_lotes').html(html);
            }
          });
        }
      });

      // 3. Buscar recetas por cédula
      $('#cedula_busqueda').on('keyup', function() {
        if (this.value.length > 5) {
          $.ajax({
            url: '../../cfg/ajax/obtener_prescripciones.php',
            type: 'POST', data: { busqueda: this.value },
            success: function(res) { $('#id_prescripcion').html(res); }
          });
        }
      });

      // 4. CARGAR TABLA DE RECETA CON LOTES INDEPENDIENTES
      $('#id_prescripcion').on('change', function() {
        const idPres = $(this).val();
        if(idPres) {
          $('#tabla_despacho_wrapper').show();
          $.ajax({
            url: '../../cfg/ajax/obtener_detalle_receta_multi.php',
            type: 'POST', data: { id_prescripcion: idPres },
            success: function(html) { 
                $('#contenedor_items_receta').html(html); 
            }
          });
        }
      });
    });
  </script>
